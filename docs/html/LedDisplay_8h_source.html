102  void writeDecimalPointAt(uint8_t pos, bool state = true) {
103  if (pos >= mLedModule.getNumDigits()) return;
104  uint8_t pattern = mLedModule.getPatternAt(pos);
105  if (state) {
106  pattern |= 0x80;
107  } else {
108  pattern &= ~0x80;
109  }
110  mLedModule.setPatternAt(pos, pattern);
111  }
